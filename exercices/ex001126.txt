\exercice{1126, liousse, 2003/10/01}

\enonce
 Montrer que 
\[
\left| 
\begin{array}{ccc}
\cos a & \cos b & \cos c \\ 
\sin a & \sin b & \sin c \\ 
1 & 1 & 1
\end{array}
\right| =\sin \left( c-b\right) +\sin \left( b-a\right) +\sin \left(
a-c\right) =4\sin \frac{c-b}{2}\sin \frac{b-a}{2}\sin \frac{a-c}{2} 
\] 
\finenonce

\noindication

\nocorrection

\finexercice
