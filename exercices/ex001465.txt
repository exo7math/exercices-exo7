\exercice{1465, ridde, 1999/11/01}

\enonce
Soient $a_{1}, \ldots, a_{n},b_{1}, \ldots, b_{n}, c_{1}, \ldots, c_{n} $ des
r\'eels positifs.\\
Montrer que $\sum\limits_{k = 1}^{n}{a_{k}b_{k}c_{k}} \leq
 \sum\limits_{k = 1}^{n}{a_{k}^{2}c_{k}} \sum\limits_{k = 1}^{n}{b_{k}^{2}c_{k}}$.
\finenonce

\noindication

\nocorrection

\finexercice
