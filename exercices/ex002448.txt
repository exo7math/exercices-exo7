\exercice{2448, matexo1, 2002/02/01}

\enonce
Calculer les d\'eterminants suivants
$$\left|
\begin{array}{cc}
4 & 5 \\
-7 & 8
\end{array}
\right|,
\qquad
\left|
\begin{array}{ccc}
1 & 3 & 2 \\
-1 & -2 & 3 \\
4 & 0 & 1
\end{array}
\right|,
\qquad
\left|
\begin{array}{cccc}
1 & 0 & 3 & 2 \\
-1 & 1 & 8 &-5 \\ 
2 & 0 & 6 & 3 \\
-1 & 0 & -3 & -7
\end{array}
\right|.$$
\finenonce

\noindication
\nocorrection

\finexercice
