\exercice{982, cousquer, 2003/10/01}

\enonce
 On considère dans $\mathbb{R}^4$, $F=\mbox{lin}\{a,b,c\}$ et $G=\mbox{lin}\{d,e\}$,
avec $a=(1,2,3,4)$, $b=(2,2,2,6)$, $c=(0,2,4,4)$, $d=(1,0,-1,2)$ et
$e=(2,3,0,1)$. Déterminer des bases des sous-espaces $F\cap G$, $F$,
$G$, $F+G$.
\finenonce

\noindication

\nocorrection

\finexercice
